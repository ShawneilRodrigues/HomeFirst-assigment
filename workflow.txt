## **File 5: `app/tools.py`**

Custom Python functions used as Agno tools. The LLM never does arithmetic — it calls these deterministically. See [Agent Tools](/tools/agent) and [Python Functions as Tools](/tools/creating-tools/python-functions).

```python
"""Deterministic tool functions — the LLM NEVER does arithmetic."""
import json
from agno.run import RunContext
from app.config import DEFAULT_INTEREST_RATE, FOIR_LIMIT


def calculate_emi(
    run_context: RunContext,
    principal: float,
    annual_rate: float,
    tenure_months: int,
) -> str:
    """Calculate EMI for a home loan.

    Args:
        principal: Loan principal amount in INR.
        annual_rate: Annual interest rate as a percentage (e.g., 12.0).
        tenure_months: Loan tenure in months.

    Returns:
        str: JSON with emi, total_payable, total_interest or error.
    """
    if principal <= 0 or annual_rate <= 0 or tenure_months <= 0:
        return json.dumps({"error": "All inputs must be positive numbers."})

    r = annual_rate / 12 / 100  # monthly rate
    n = tenure_months
    emi = principal * r * ((1 + r) ** n) / (((1 + r) ** n) - 1)
    total_payable = emi * n
    total_interest = total_payable - principal

    result = {
        "emi": round(emi, 2),
        "total_payable": round(total_payable, 2),
        "total_interest": round(total_interest, 2),
    }

    # Update session state
    run_context.session_state["tool_called"] = True
    run_context.session_state["eligibility_result"] = result

    return json.dumps(result)


def check_loan_eligibility(
    run_context: RunContext,
    monthly_income: float,
    property_value: float,
    loan_requested: float,
    existing_emis: float,
    employment_status: str,
    tenure_months: int,
    annual_rate: float = DEFAULT_INTEREST_RATE,
) -> str:
    """Check home loan eligibility based on FOIR and LTV norms.

    Args:
        monthly_income: Applicant's net monthly income in INR.
        property_value: Property value in INR.
        loan_requested: Loan amount requested in INR.
        existing_emis: Total existing monthly EMI obligations in INR.
        employment_status: 'salaried' or 'self_employed'.
        tenure_months: Desired loan tenure in months.
        annual_rate: Annual interest rate (default from config).

    Returns:
        str: JSON with status, eligible_amount, reason, recommended_emi, foir_used, ltv_cap.
    """
    # Step 1 — FOIR check
    available_emi = monthly_income * FOIR_LIMIT - existing_emis
    if available_emi <= 0:
        result = {
            "status": "REJECTED",
            "eligible_amount": 0,
            "reason": "FOIR exceeded — existing EMIs consume more than 50% of income.",
            "recommended_emi": 0,
            "foir_used": round((existing_emis / monthly_income) * 100, 2),
            "ltv_cap": 0,
        }
        run_context.session_state["tool_called"] = True
        run_context.session_state["eligibility_result"] = result
        return json.dumps(result)

    # Step 2 — LTV cap
    if property_value < 3000000:  # < 30L
        max_ltv = 0.90
    elif property_value <= 7500000:  # 30L–75L
        max_ltv = 0.80
    else:  # > 75L
        max_ltv = 0.75

    max_loan_ltv = property_value * max_ltv

    # Step 3 — Max loan from FOIR (reverse EMI calculation)
    r = annual_rate / 12 / 100
    n = tenure_months
    if r > 0 and n > 0:
        max_loan_foir = available_emi * (((1 + r) ** n) - 1) / (r * ((1 + r) ** n))
    else:
        max_loan_foir = 0

    # Step 4 — Final eligibility
    eligible_amount = min(max_loan_ltv, max_loan_foir, loan_requested)

    if eligible_amount >= loan_requested:
        status = "APPROVED"
        reason = "Full loan amount is within FOIR and LTV limits."
    elif eligible_amount > 0:
        status = "PARTIAL"
        reason = f"Eligible for ₹{eligible_amount:,.0f} instead of requested ₹{loan_requested:,.0f}."
    else:
        status = "REJECTED"
        reason = "Loan amount exceeds both FOIR and LTV limits."

    # Calculate recommended EMI
    if eligible_amount > 0:
        rec_emi = eligible_amount * r * ((1 + r) ** n) / (((1 + r) ** n) - 1)
    else:
        rec_emi = 0

    result = {
        "status": status,
        "eligible_amount": round(eligible_amount, 2),
        "reason": reason,
        "recommended_emi": round(rec_emi, 2),
        "foir_used": round(((existing_emis + rec_emi) / monthly_income) * 100, 2) if monthly_income > 0 else 0,
        "ltv_cap": max_ltv * 100,
    }

    # Update session state
    run_context.session_state["tool_called"] = True
    run_context.session_state["eligibility_result"] = result

    return json.dumps(result)
```

```python
"""Deterministic lead scoring — runs in Python, not the LLM."""
from app.config import LEAD_SCORE_HANDOFF_THRESHOLD


def compute_lead_score(entity_state: dict, user_message: str = "") -> int:
    """Compute lead score (0–10) based on entity completeness and intent signals."""
    score = 0

    # +3 if all 4 core entities are present
    core_entities = ["monthly_income", "property_value", "loan_amount_requested", "employment_status"]
    if all(entity_state.get(e) for e in core_entities):
        score += 3

    # +3 if eligibility tool returned APPROVED
    elig = entity_state.get("eligibility_result")
    if elig and isinstance(elig, dict) and elig.get("status") == "APPROVED":
        score += 3

    # +2 for high-intent phrases
    high_intent_phrases = [
        "when can i apply", "how to proceed", "kab milega loan",
        "next step", "aage kya karna hai", "apply karna hai",
        "documents bhejun", "ready to apply",
    ]
    msg_lower = user_message.lower()
    if any(phrase in msg_lower for phrase in high_intent_phrases):
        score += 2

    # +1 for document-related questions
    doc_phrases = ["documents", "dastavez", "kya chahiye", "papers", "kagaz"]
    if any(phrase in msg_lower for phrase in doc_phrases):
        score += 1

    # +1 if tenure preference provided
    if entity_state.get("tenure_months"):
        score += 1

    return min(score, 10)


def should_handoff(score: int) -> bool:
    return score >= LEAD_SCORE_HANDOFF_THRESHOLD
```

Ties together STT → Agent → Lead Scoring → TTS.

```python
"""Orchestrator — glues STT, Agent, Lead Scoring, and TTS together."""
import asyncio
from app.counselor_agent import create_counselor_agent
from app.stt_client import transcribe_audio
from app.tts_client import synthesize_speech
from app.lead_scorer import compute_lead_score, should_handoff


class VoiceOrchestrator:
    def __init__(self):
        self.agent = create_counselor_agent()

    async def process_voice_turn(
        self, audio_bytes: bytes, session_id: str, user_id: str = "anonymous"
    ) -> dict:
        """Process one voice turn: STT → Agent → Lead Score → TTS.

        Returns:
            dict with keys: transcript, response_text, audio_bytes, entity_state,
                            lead_score, handoff_triggered, tool_called
        """
        # 1. STT — transcribe audio
        current_lang = self.agent.session_state.get("locked_language") or "hi"
        lang_map = {"hi": "hi-IN", "en": "en-IN", "mr": "mr-IN", "ta": "ta-IN"}
        stt_result = await transcribe_audio(
            audio_bytes, language_code=lang_map.get(current_lang, "hi-IN")
        )

        transcript = stt_result["transcript"]
        confidence = stt_result["confidence"]

        # Confidence gating — ask user to repeat if STT is uncertain
        if confidence < 0.7:
            fallback_text = {
                "hi": "Maaf kijiye, mujhe aapki baat samajh nahi aayi. Kya aap dobara bol sakte hain?",
                "en": "Sorry, I didn't catch that. Could you please repeat?",
                "mr": "Maaf kara, mala samajale nahi. Krupaya punha sanga.",
                "ta": "Mannikkavum, enakku puriyavillai. Thayavu seithu meendum sollungal.",
            }.get(current_lang, "Sorry, I didn't catch that. Could you please repeat?")

            audio = await synthesize_speech(fallback_text, current_lang)
            return {
                "transcript": transcript,
                "response_text": fallback_text,
                "audio_bytes": audio,
                "entity_state": self.agent.session_state,
                "lead_score": self.agent.session_state.get("lead_score", 0),
                "handoff_triggered": False,
                "tool_called": False,
                "confidence": confidence,
            }

        # 2. Language lock — set on first turn if not already locked
        if not self.agent.session_state.get("locked_language"):
            detected_lang = stt_result["language_code"][:2]
            self.agent.session_state["locked_language"] = detected_lang

        # 3. Run the Agno Agent
        response = self.agent.run(
            transcript,
            session_id=session_id,
            user_id=user_id,
        )
        response_text = response.content

        # 4. Update turn count
        state = self.agent.get_session_state() or {}
        state["turn_count"] = state.get("turn_count", 0) + 1

        # 5. Lead scoring (deterministic, not LLM)
        lead_score = compute_lead_score(state, transcript)
        state["lead_score"] = lead_score
        handoff = should_handoff(lead_score)
        state["handoff_triggered"] = handoff

        # 6. TTS — synthesize response
        locked_lang = state.get("locked_language", "hi")
        audio = await synthesize_speech(response_text, locked_lang)

        return {
            "transcript": transcript,
            "response_text": response_text,
            "audio_bytes": audio,
            "entity_state": state,
            "lead_score": lead_score,
            "handoff_triggered": handoff,
            "tool_called": state.get("tool_called", False),
            "confidence": confidence,
        }
```

```python
"""FastAPI entry point with AgentOS integration."""
from fastapi import FastAPI, UploadFile, File, Form
from fastapi.responses import JSONResponse
import asyncio
import base64

from agno.os import AgentOS
from app.counselor_agent import create_counselor_agent
from app.orchestrator import VoiceOrchestrator

# Custom FastAPI app
app = FastAPI(title="HomeFirst Loan Counselor API", version="1.0.0")

orchestrator = VoiceOrchestrator()


@app.get("/health")
async def health():
    return {"status": "healthy", "service": "homefirst-loan-counselor"}


@app.post("/api/voice")
async def voice_endpoint(
    audio: UploadFile = File(...),
    session_id: str = Form(...),
    user_id: str = Form(default="anonymous"),
):
    """Main voice endpoint: receives audio, returns response + audio."""
    audio_bytes = await audio.read()

    result = await orchestrator.process_voice_turn(
        audio_bytes=audio_bytes,
        session_id=session_id,
        user_id=user_id,
    )

    # Encode audio as base64 for JSON transport
    audio_b64 = base64.b64encode(result["audio_bytes"]).decode("utf-8")

    return JSONResponse({
        "transcript": result["transcript"],
        "response_text": result["response_text"],
        "audio_base64": audio_b64,
        "entity_state": result["entity_state"],
        "lead_score": result["lead_score"],
        "handoff_triggered": result["handoff_triggered"],
        "tool_called": result["tool_called"],
        "confidence": result["confidence"],
    })


@app.post("/api/text")
async def text_endpoint(
    message: str = Form(...),
    session_id: str = Form(...),
    user_id: str = Form(default="anonymous"),
):
    """Text-only endpoint for testing without voice."""
    agent = orchestrator.agent
    response = agent.run(message, session_id=session_id, user_id=user_id)
    state = agent.get_session_state() or {}

    from app.lead_scorer import compute_lead_score, should_handoff
    score = compute_lead_score(state, message)

    return JSONResponse({
        "response_text": response.content,
        "entity_state": state,
        "lead_score": score,
        "handoff_triggered": should_handoff(score),
    })


# Integrate with AgentOS for built-in agent endpoints
agent_os = AgentOS(
    agents=[create_counselor_agent()],
    base_app=app,
)
app = agent_os.get_app()
```

```python
        st.session_state.messages.append({"role": "user", "content": user_input})
        with st.chat_message("user"):
            st.write(user_input)

        # Call text API
        with st.spinner("Priya is thinking..."):
            resp = httpx.post(
                f"{API_URL}/api/text",
                data={
                    "message": user_input,
                    "session_id": st.session_state.session_id,
                    "user_id": "streamlit_user",
                },
                timeout=30.0,
            )
            data = resp.json()

        response_text = data["response_text"]
        st.session_state.messages.append({"role": "assistant", "content": response_text})
        st.session_state.entity_state = data.get("entity_state", {})
        st.session_state.lead_score = data.get("lead_score", 0)

        with st.chat_message("assistant"):
            st.write(response_text)

        if data.get("handoff_triggered"):
            st.error("🚨 HANDOFF TRIGGERED: Routing to Human RM")

with right_col:
    st.subheader("🔍 Debug Panel")

    state = st.session_state.entity_state

    # Language
    lang = state.get("locked_language", "Not detected")
    st.metric("Language", f"{lang} — {'LOCKED' if lang else 'detecting...'}")

    # Lead Score
    score = st.session_state.lead_score
    color = "🔴" if score < 5 else "🟠" if score < 8 else "🟢"
    st.metric("Lead Score", f"{color} {score}/10")
    st.progress(score / 10)

    # Entity State
    st.subheader("Extracted Entities")
    entity_display = {
        "Monthly Income": state.get("monthly_income"),
        "Property Value": state.get("property_value"),
        "Loan Requested": state.get("loan_amount_requested"),
        "Employment": state.get("employment_status"),
        "Existing EMIs": state.get("existing_emis"),
        "Tenure (months)": state.get("tenure_months"),
    }
    for k, v in entity_display.items():
        st.text(f"{k}: {v or '—'}")

    # Tool info
    st.subheader("Tool Execution")
    st.text(f"Tool Called: {state.get('tool_called', False)}")
    if state.get("eligibility_result"):
        st.json(state["eligibility_result"])

    # Handoff
    if state.get("handoff_triggered"):
        st.error("🚨 HANDOFF TRIGGERED")
```
